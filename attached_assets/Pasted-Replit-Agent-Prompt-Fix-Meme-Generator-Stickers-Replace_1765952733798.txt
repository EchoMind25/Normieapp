jsx// client/components/MemeGenerator/StickerPanel.jsx

import { normieStickers, cryptoStickers, brandStickers } from './stickerConfig';

function StickerPanel({ onStickerSelect }) {
  const [activeTab, setActiveTab] = useState('normie');
  const [imageLoadErrors, setImageLoadErrors] = useState({});

  const getStickersByTab = () => {
    switch(activeTab) {
      case 'normie': return normieStickers;
      case 'crypto': return cryptoStickers;
      case 'brand': return brandStickers;
      default: return [];
    }
  };

  const handleImageError = (stickerId) => {
    console.error(`Failed to load sticker: ${stickerId}`);
    setImageLoadErrors(prev => ({ ...prev, [stickerId]: true }));
  };

  return (
    <div className="sticker-panel">
      <div className="sticker-tabs">
        <button 
          className={activeTab === 'normie' ? 'active' : ''} 
          onClick={() => setActiveTab('normie')}
        >
          Normie
        </button>
        <button 
          className={activeTab === 'crypto' ? 'active' : ''} 
          onClick={() => setActiveTab('crypto')}
        >
          Crypto
        </button>
        <button 
          className={activeTab === 'brand' ? 'active' : ''} 
          onClick={() => setActiveTab('brand')}
        >
          Brand
        </button>
      </div>

      <div className="sticker-size-control">
        <label>Sticker Size: 1.5x</label>
        <input type="range" min="0.5" max="3" step="0.1" defaultValue="1.5" />
      </div>

      <div className="sticker-grid">
        {getStickersByTab().map(sticker => (
          <div 
            key={sticker.id} 
            className="sticker-item"
            onClick={() => !imageLoadErrors[sticker.id] && onStickerSelect(sticker)}
          >
            {imageLoadErrors[sticker.id] ? (
              <div className="sticker-error">
                <span>‚ùå</span>
                <p>Missing</p>
              </div>
            ) : (
              <>
                <img 
                  src={sticker.url} 
                  alt={sticker.name}
                  onError={() => handleImageError(sticker.id)}
                  loading="lazy"
                />
                <span className="sticker-name">{sticker.name}</span>
              </>
            )}
          </div>
        ))}
      </div>

      {Object.keys(imageLoadErrors).length > 0 && (
        <div className="sticker-warning">
          ‚ö†Ô∏è Some stickers failed to load. Please add images to /public/stickers/
        </div>
      )}
    </div>
  );
}

export default StickerPanel;

const addStickerToCanvas = (sticker) => {

    sticker.url,
    (img) => {
      // Set initial size based on slider
      const scale = stickerSize || 1.5;
      img.scale(scale);

      // Center on canvas

        left: canvas.width / 2,

        originX: 'center',

      });

      // Make it selectable and movable
      img.setControlsVisibility({
        mt: true, // middle top
        mb: true, // middle bottom
        ml: true, // middle left
        mr: true, // middle right
        bl: true, // bottom left
        br: true, // bottom right
        tl: true, // top left
        tr: true, // top right
        mtr: true // rotating point

      canvas.add(img);
      canvas.setActiveObject(img);
      canvas.renderAll();

    }

javascript// scripts/generate-brand-stickers.js
const sharp = require('sharp');

async function createTextSticker(text, filename) {
  const svg = `
    <svg width="400" height="400">
      <rect width="400" height="400" fill="black" rx="20"/>
      <text 
        x="50%" 
        y="50%" 
        font-family="Impact, Arial Black" 
        font-size="60" 
        font-weight="bold" 
        fill="#00ff00" 
        text-anchor="middle" 
        dominant-baseline="middle"
      >
        ${text}
      </text>
    </svg>
  `;
  
  await sharp(Buffer.from(svg))
    .png()
    .toFile(`public/stickers/brand/${filename}.png`);
}

async function generateBrandStickers() {
  await createTextSticker('NORMIE', 'normie-text');
  await createTextSticker('WAGMI', 'wagmi-stamp');
  await createTextSticker('NGMI', 'ngmi-stamp');
  await createTextSticker('BASED', 'based-stamp');
  await createTextSticker('572M\nBURNED', 'burn-badge');
  await createTextSticker('230M\nLOCKED', 'lock-badge');
  
  console.log('Brand stickers generated!');
}
